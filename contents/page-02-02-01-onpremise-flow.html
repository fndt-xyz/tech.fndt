mermaid source code (optimized for diagrams.net):

graph TD
    subgraph Office_HQ [open.fndt.xyz office - static public IP]
        TS["tunneling server (xubuntu 24.04)"]
        Dev["admin / maintenance PC"]
    end

    subgraph client_site [client premise - local LAN]
        ERP["ERP server mini-PC (xubuntu / 16GB RAM / docker)"]
        User["local employee devices"]
        Router["client router (no port forwarding req.)"]
    end

    subgraph External [remote access]
        RemoteUser["client employee @ home"]
    end

    %% Flow logic
    User -- "local LAN access" --> ERP
    ERP -- "initiates outbound tunnel" --> TS
    TS -- "handshake/bridge" --> ERP

    %% Admin logic
    Dev -- "remote maintenance" --> TS
    TS -- "secure tunnel" --> ERP

    %% Remote user logic
    RemoteUser -- "requests ERP web UI" --> TS
    TS -- "routes traffic" --> ERP
